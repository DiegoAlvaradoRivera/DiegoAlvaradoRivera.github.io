fit_measure <- function(model){
  aic <- AIC(model)                    # AIC
  gcv <- summary(model)$sp.criterion   # GCV
  rsq <- summary(model)$r.sq           # R squared
  
  cat("AIC:", aic, " GCV:", gcv, " R.SQ:", rsq, "\n")
}
